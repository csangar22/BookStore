        // Aquí se cargará dinámicamente la información del libro seleccionado
        $(document).ready(function() {
            // Obtener los parámetros de la URL
            const urlParams = new URLSearchParams(window.location.search);
            const bookTitle = urlParams.get('title');
            const bookAuthor = urlParams.get('author');
            const bookStars = urlParams.get('stars');
            const bookPrice = urlParams.get('price');
            const bookImage = urlParams.get('image');

            // Actualizar el contenido de la página con los detalles del libro
            $('#book-image img').attr('src', bookImage);
            $('#book-title').text(bookTitle);
            $('#book-author').text(bookAuthor);
            $('#book-stars').html(generateStars(bookStars));
            $('#book-price').text(bookPrice + ' €');

            // Función para generar estrellas
            function generateStars(starCount) {
                let starsHtml = '';
                for (let i = 0; i < 5; i++) {
                    if (i < Math.floor(starCount)) {
                        starsHtml += '&#9733;'; // Estrella llena
                    } else if (i === Math.floor(starCount) && starCount % 1 !== 0) {
                        starsHtml += '&#9733;&#189;'; // Media estrella
                    } else {
                        starsHtml += '&#9734;'; // Estrella vacía
                    }
                }
                return starsHtml;
            }
        });
    </script>
